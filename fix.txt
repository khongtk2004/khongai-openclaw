cd ~/khongai-openclaw && \
docker build -t khongai-openclaw:latest . && \
mkdir -p ~/khongai && cd ~/khongai && \
cat > docker-compose.yml << 'EOF'
version: '3.8'
services:
  khongai:
    image: khongai-openclaw:latest
    container_name: khongai
    restart: unless-stopped
    volumes:
      - ~/.khongai:/home/node/.khongai
      - ~/.khongai/workspace:/home/node/.khongai/workspace
    ports:
      - "18789:18789"
    environment:
      - NODE_ENV=production
      - TELEGRAM_BOT_TOKEN=YourTokenBottelegram
      - ADMIN_USER_IDS=YourUserNameTelegram
    command: ["gateway", "start", "--foreground"]
EOF
mkdir -p ~/.khongai/workspace && \
docker compose up -d && \
docker ps && \
echo "✅ KhongAI is running! Check logs with: docker logs khongai"
